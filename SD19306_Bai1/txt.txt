- Tiêu: dễ lỗi, một khi lỗi thì cực khó sửa

-- công cụ cần cho lập trình winform
-- 1.toolbox: thanh công cụ View
-- 2.Properties: khi chọn vào biến trên màn hình


-- các biến:
-- 1.label: text trên màn hình
     + text: sửa chữ hiển thị
-- 2.textbox: dùng để nhập, hiển thị
   3.richtextbox: dùng để nhập, hiển thị nhưng lực hơn
   4.radiobutton: dùng để chọn những lựa chọn cứng
   5.checkbox: dùng để multiple choice

   -- properties: Text: hiển thị lên màn hình
                  name
-- QUY TẮC ĐẶT TÊN:
textbox:        txt_<tên> (VD: txt_MSV, txt_hoten)
richtextbox:    rtxt_<tên>
radiobutton:    rbtn_<tên>
checkbox:       cbx_<tên>
button:         btn_<tên>
datagridview:   dtgView

datepicker:     dpk
combobox:       cmbx_<tên>
